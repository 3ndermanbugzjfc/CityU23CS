#include <bits/stdc++.h>
using namespace std;
const int N = 350;
int n, m, s, u, v, a[4];
long long ans, cnt[N], cntbad[N][N];
bool hv[N][N];
vector<int> adj[N], goodadj[N], pairs[N][N];
vector<pair<int, int>> edgelist, goodlist;
void solve() {
    cin >> n >> m >> s;
    ans = 0;
    for (int i = 0; i <= n; i++)
        for (int j = 0; j <= n; j++) hv[i][j] = i == j, pairs[i][j].clear(), cntbad[i][j] = 0;
    for (int i = 0; i <= n; i++) adj[i].clear(), goodadj[i].clear(), cnt[i] = 0;
    edgelist.clear();
    goodlist.clear();
    for (int i = 1; i <= m; i++) {
        cin >> u >> v;
        hv[u][v] = hv[v][u] = 1;
        adj[u].push_back(v);
        adj[v].push_back(u);
        edgelist.push_back({ u, v });
    }
    for (auto i : edgelist) {
        if (hv[s][i.first] || hv[s][i.second])
            continue;
        goodlist.push_back(i);
        goodadj[i.first].push_back(i.second);
        goodadj[i.second].push_back(i.first);
    }
    for (int i = 1; i <= n; i++) {
        if (i == s)
            continue;
        for (int j = 0; j < goodadj[i].size(); j++) {
            for (int k = 0; k < j; k++) {
                int x1 = goodadj[i][j], x2 = goodadj[i][k];
                if (x1 > x2)
                    swap(x1, x2);
                pairs[x1][x2].push_back(i);
                // cout<<x1<<"<-"<<i<<"->"<<x2<<"\n";
            }
        }
    }
    for (int i = 1; i <= n; i++) {
        for (int j = i + 1; j <= n; j++) {
            if (i == j || i == s || j == s)
                continue;
            for (int a1 : pairs[i][j])
                for (int a2 : pairs[i][j])
                    if (a1 < a2) {
                        // i a1 j a2 is cycle
                        a[0] = i, a[1] = a1, a[2] = j, a[3] = a2;
                        for (int id = 0; id < 4; id++) {
                            cnt[a[id]]++;
                            // cout<<a[id]<<" ";
                            for (int di = 0; di < 4; di++) {
                                if (id == di)
                                    continue;
                                cntbad[a[id]][a[di]]++;
                            }
                        }
                        // cout<<endl;
                    }
        }
    }
    for (int i : adj[s]) {
        for (int j : adj[i]) {
            if (hv[s][j])
                continue;
            for (int k : adj[j]) ans += cnt[k] - cntbad[k][j];
        }
    }
    cout << (ans / 2) % 1000000007 << "\n";
}
int main() {
    int T;
    cin >> T;
    while (T--) solve();
}
