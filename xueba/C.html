#include <bits/stdc++.h>
using namespace std;
const int N = 1550;
typedef long long ll;
const ll MOD = 1000000007;
ll n, m, a[N][N], ans;
ll calc(int x, int y) {
    if (x <= 0 || y <= 0)
        return 0;
    ll ret = 0;
    ret += a[n][m] * (x / n) % MOD * (y / m) % MOD;
    ret += a[x % n][m] * (y / m) % MOD;
    ret += a[n][y % m] * (x / n) % MOD;
    ret += a[x % n][y % m] % MOD;
    return ret;
}
int main() {
    cin >> n >> m;
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= m; j++) {
            cin >> a[i][j];
            a[i][j] += a[i - 1][j];
            a[i][j] += a[i][j - 1];
            a[i][j] -= a[i - 1][j - 1];
            a[i][j] %= MOD;
        }
    }
    int x, y, xa, ya;
    cin >> x >> xa >> y >> ya;
    // x--,xa--,y--,ya--;
    cout << ((calc(xa, ya) - calc(x - 1, ya) - calc(xa, y - 1) + calc(x - 1, y - 1)) % MOD + MOD) % MOD
         << "\n";
}
